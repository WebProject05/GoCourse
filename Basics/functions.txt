package main

import "fmt"

func main() {
	// func <name>{para list}( doing some things)
	value := add(5, 5)
	fmt.Println(value)

	func() {	// These are anonymaus functions
		fmt.Println("hi mi")
	}()
	
	greet := func() {
		fmt.Println("hi mi")
	}

	greet()

	result := applyOperations(1,2, operation)
	fmt.Println(result)

}

func add(a, b int) int {
	return a + b
}

// Passing functions into the function parameters
func applyOperations(x int, y int, operation func(int, int) int) int {
	return operation(x,y)
}

func operation(a int, b int) int {
	return a + b
}
