package main

import "fmt"

func main() {
	// panic(interface{}) the interface meaning that the input can be of any type

	process(20)
	fmt.Println("---")
	process(-20)
}

// after all the defer funcions have exceuted the panic executes
func process(input int) {

	defer fmt.Println("Deffered 1")
	defer fmt.Println("Deffered 2")
	if input < 0 {
		fmt.Println("Before Panic")
		panic("Input must be non-negative")
		// fmt.Println("After panic") This part of the code is unreachable
	}
	fmt.Println("Processing the input")
}
