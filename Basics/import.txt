package main

import "fmt"
import "net/http"

func main() {
	fmt.Println("Hello! GoLang")

	resp, err := http.Get("https://jsonplaceholder.typicode.com/posts/1")
	if err != nil {
		fmt.Println("Error: ", err)
		return
	}

	defer resp.Body.Close()

	fmt.Println("Http Response code is: ", resp.Status)
}
